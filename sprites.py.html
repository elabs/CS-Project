110     def start_move(self,e):
111         if e.key in self.controls.keys():
112             movement = self.controls[e.key]
113             self.moving = True
114             if movement == 'left':
115                 self.physics.acc.x = -self.settings.base_acc
116                 self.last_state = 'left'
117             if movement == 'right':
118                 self.last_state = 'right'
119                 self.physics.acc.x = +self.settings.base_acc
120 
121     def stop_move(self,e):
122         try:
123             if self.controls[e.key] == self.last_state:
124                 self.moving = False
125                 self.physics.acc = vec(0,0)
126             if self.controls[e.key] == 'down':
127                 self.physics.acc.y = 0
128         except:
129             pass
